```sql
-- ==========================================================
-- SURSA SQL 1: INFRASTRUCTURĂ TRANSPORT (Oracle)
-- ==========================================================

-- 1. Tabelul pentru stații
CREATE TABLE BUS_STOPS (
    STOP_ID NUMBER PRIMARY KEY,
    STOP_NAME VARCHAR2(500),
    STOP_LAT NUMBER,
    STOP_LON NUMBER
);

-- 2. Tabelul pentru rute
CREATE TABLE BUS_ROUTES (
    ROUTE_ID VARCHAR2(200) PRIMARY KEY, 
    ROUTE_SHORT_NAME VARCHAR2(200),
    ROUTE_LONG_NAME VARCHAR2(1000), 
    ROUTE_TYPE VARCHAR2(100)
);

-- 3. Tabelul de legătură 
CREATE TABLE ROUTE_STOPS (
    ROUTE_ID VARCHAR2(200),
    STOP_ID NUMBER,
    STOP_SEQUENCE NUMBER,
 
    CONSTRAINT FK_ROUTE FOREIGN KEY (ROUTE_ID) REFERENCES BUS_ROUTES(ROUTE_ID),
    CONSTRAINT FK_STOP FOREIGN KEY (STOP_ID) REFERENCES BUS_STOPS(STOP_ID)
);

-- ==========================================================
-- SURSA DOCUMENT: STATISTICI ANALITICE (CSV Import)
-- ==========================================================


CREATE TABLE TRANSPORT_STATS (
    STAT_ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    REPORT_MONTH VARCHAR2(100), -- Importat ca text pentru a evita conflictele de format DATE
    TRANSIT_RIDERSHIP NUMBER,   -- Volum pasageri
    GASOLINE_PRICE NUMBER,      -- Prețul combustibilului (corelație)
    ON_TIME_PERF NUMBER         -- Eficiența serviciului
);
```
